import java.util.Arrays;
import java.util.Random;

public class SortingComparison {
    private static final int ARRAY_SIZE = 10000;
    private static final int WORD_LENGTH = 5;
    
    public static void main(String[] args) {
        // Generate random words
        String[] originalArray = generateRandomWords(ARRAY_SIZE);
        
        // Test each sorting algorithm
        testSort("QuickSort", originalArray.clone(), SortingComparison::quickSort);
        testSort("MergeSort", originalArray.clone(), SortingComparison::mergeSort);
        testSort("HeapSort", originalArray.clone(), SortingComparison::heapSort);
        testSort("Built-in Arrays.sort", originalArray.clone(), Arrays::sort);
    }
    
    // Interface for sorting method
    interface SortingMethod {
        void sort(String[] arr);
    }
    
    // Test harness for sorting algorithms
    private static void testSort(String name, String[] arr, SortingMethod sortMethod) {
        long startTime = System.nanoTime();
        sortMethod.sort(arr);
        long endTime = System.nanoTime();
        
        System.out.printf("%s took %.3f ms%n", 
            name, (endTime - startTime) / 1_000_000.0);
        
        // Verify sorting
        System.out.println("Correctly sorted: " + isSorted(arr));
    }
    
    // Random word generator
    private static String[] generateRandomWords(int count) {
        String[] words = new String[count];
        Random random = new Random();
        
        for (int i = 0; i < count; i++) {
            StringBuilder word = new StringBuilder();
            for (int j = 0; j < WORD_LENGTH; j++) {
                char c = (char) (random.nextInt(26) + 'a');
                word.append(c);
            }
            words[i] = word.toString();
        }
        return words;
    }
    
    // QuickSort implementation
    public static void quickSort(String[] arr) {
        quickSort(arr, 0, arr.length - 1);
    }
    
    private static void quickSort(String[] arr, int low, int high) {
        if (low < high) {
            int pi = partition(arr, low, high);
            quickSort(arr, low, pi - 1);
            quickSort(arr, pi + 1, high);
        }
    }
    
    private static int partition(String[] arr, int low, int high) {
        String pivot = arr[high];
        int i = low - 1;
        
        for (int j = low; j < high; j++) {
            if (arr[j].compareTo(pivot) <= 0) {
                i++;
                String temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
        
        String temp = arr[i + 1];
        arr[i + 1] = arr[high];
        arr[high] = temp;
        
        return i + 1;
    }
    
    // MergeSort implementation
    public static void mergeSort(String[] arr) {
        if (arr.length <= 1) return;
        
        int mid = arr.length / 2;
        String[] left = Arrays.copyOfRange(arr, 0, mid);
        String[] right = Arrays.copyOfRange(arr, mid, arr.length);
        
        mergeSort(left);
        mergeSort(right);
        merge(arr, left, right);
    }
    
    private static void merge(String[] arr, String[] left, String[] right) {
        int i = 0, j = 0, k = 0;
        
        while (i < left.length && j < right.length) {
            if (left[i].compareTo(right[j]) <= 0) {
                arr[k++] = left[i++];
            } else {
                arr[k++] = right[j++];
            }
        }
        
        while (i < left.length) arr[k++] = left[i++];
        while (j < right.length) arr[k++] = right[j++];
    }
    
    // HeapSort implementation
    public static void heapSort(String[] arr) {
        int n = arr.length;
        
        // Build heap
        for (int i = n / 2 - 1; i >= 0; i--)
            heapify(arr, n, i);
            
        // Extract elements from heap
        for (int i = n - 1; i >= 0; i--) {
            String temp = arr[0];
            arr[0] = arr[i];
            arr[i] = temp;
            
            heapify(arr, i, 0);
        }
    }
    
    private static void heapify(String[] arr, int n, int i) {
        int largest = i;
        int left = 2 * i + 1;
        int right = 2 * i + 2;
        
        if (left < n && arr[left].compareTo(arr[largest]) > 0)
            largest = left;
            
        if (right < n && arr[right].compareTo(arr[largest]) > 0)
            largest = right;
            
        if (largest != i) {
            String swap = arr[i];
            arr[i] = arr[largest];
            arr[largest] = swap;
            
            heapify(arr, n, largest);
        }
    }
    
    // Utility method to verify sorting
    private static boolean isSorted(String[] arr) {
        for (int i = 0; i < arr.length - 1; i++) {
            if (arr[i].compareTo(arr[i + 1]) > 0) {
                return false;
            }
        }
        return true;
    }
}
